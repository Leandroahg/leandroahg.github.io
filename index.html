document.getElementById('btn-pagamento').addEventListener('click', async () => { try { // Autenticação necessária antes do pagamento await Pi.authenticate(['payments'], { onIncompletePaymentFound: (payment) => { console.log(payment); }, onReadyForServerCreate: (payment) => { console.log(payment); } }); // Criação do pagamento await Pi.createPayment({ amount: 0.1, memo: "Teste final", metadata: { type: "test" }, }, { onReadyForServerCreate: (paymentId) => { alert("ID Gerado: " + paymentId); }, onReadyForServerApproval: (paymentId) => { alert("Pronto para aprovação: " + paymentId); }, onReadyForServerCompletion: (paymentId, txid) => { alert("Pagamento concluído! TXID: " + txid); }, onCancel: (paymentId) => { alert("Pagamento cancelado"); }, onError: (error, paymentId) => { alert("Erro: " + error.message); } }); } catch (err) { alert("Erro: " + err.message); } });
